🔗 URL do Web App (Apps Script)
Como configurar:
1. No Google Sheets, abra Extensões → Apps Script
2. No Code.gs, adicione a função doGet() abaixo
3. Implante como Web App (Execute as: Me, Anyone)
4. Cole a URL aqui 
function doGet(e) {
  const action = e.parameter.action;
  let result;
  if (action === 'getCatalogo')       result = getCatalogo();
  else if (action === 'getColecoes')  result = getColecoes();
  else if (action === 'getMembros')   result = getMembros();
  else if (action === 'addMembro')    result = addMembro(e.parameter.nome);
  else if (action === 'removeMembro') result = removeMembro(e.parameter.nome);
  else if (action === 'toggleFlor')   result = toggleFlor(e.parameter.membro, e.parameter.florEN, e.parameter.checked === 'true');
  else if (action === 'addFlorCustom') result = addFlorCustom(e.parameter.en, e.parameter.pt, e.parameter.emoji, e.parameter.rarity);
  else result = { error: 'Ação desconhecida' };
  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}
